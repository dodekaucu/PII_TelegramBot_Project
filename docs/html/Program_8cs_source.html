150  public static async Task HandleUpdateAsync(Update update, CancellationToken cancellationToken)
151  {
152  try
153  {
154  // Sólo respondemos a mensajes de texto
155  if (update.Type == UpdateType.Message)
156  {
157  await HandleMessageReceived((new TelegramMSGadapter(update.Message)));
158  }
159  }
160  catch(Exception e)
161  {
162  await HandleErrorAsync(e, cancellationToken);
163  }
164  }

171  private static async Task HandleMessageReceived(IMessage message)
172  {
173  Console.WriteLine($"Received a message from {message.ID} saying: {message.Text}");
174 
175  string response = string.Empty;
176 
177  firstHandler.Handle(message, out response);
178 
179  if (!string.IsNullOrEmpty(response))
180  {
181  await Bot.SendTextMessageAsync(message.ID, response);
182  }
183  }
